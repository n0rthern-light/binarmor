  set_property(TARGET loader PROPERTY CXX_STANDARD 20)

  set(wxWidgets_LIB_DIR "${wxWidgets_ROOT_DIR}/lib/vc_x64_lib")  # 64-bit

  set(wxWidgets_LIB_DIR "${wxWidgets_ROOT_DIR}/lib/vc_lib")      # 32-bit
